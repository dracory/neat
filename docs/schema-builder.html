Schema Builder

This document describes the schema builder API in Neat ORM.

Creating Tables

err := db.Schema().Create("users", func(table neat.Blueprint) {
    table.ID()
    table.String("name")
    table.String("email").Unique()
    table.Timestamps()
})

Modifying Tables

err := db.Schema().Table("users", func(table neat.Blueprint) {
    table.String("phone").Nullable()
    table.DropColumn("old_column")
})

Dropping Tables

// Drop table
err := db.Schema().Drop("users")

// Drop if exists
err := db.Schema().DropIfExists("users")

Renaming Tables

err := db.Schema().Rename("users", "accounts")

Column Types

Integer Types

// Auto-incrementing integers
table.Increments("id")           // 4-byte auto-increment
table.IntegerIncrements("id")  // Alias for Increments
table.BigIncrements("id")      // 8-byte auto-increment
table.MediumIncrements("id")   // 3-byte auto-increment
table.SmallIncrements("id")   // 2-byte auto-increment
table.TinyIncrements("id")    // 1-byte auto-increment

// Regular integers
table.Integer("age")           // 4-byte integer
table.BigInteger("count")      // 8-byte integer
table.MediumInteger("status")  // 3-byte integer
table.SmallInteger("priority") // 2-byte integer
table.TinyInteger("flags")     // 1-byte integer

// Unsigned integers
table.UnsignedInteger("age")
table.UnsignedBigInteger("count")
table.UnsignedMediumInteger("status")
table.UnsignedSmallInteger("priority")
table.UnsignedTinyInteger("flags")

String Types

table.String("name")
table.String("name", 100) // with length
table.Char("code", 10)    // Fixed-length string

Text Types

table.Text("description")      // TEXT
table.TinyText("short_desc")    // TINYTEXT
table.MediumText("content")    // MEDIUMTEXT
table.LongText("article")      // LONGTEXT

Numeric Types

table.Float("price")           // FLOAT
table.Float("price", 8, 2)    // FLOAT with precision
table.Double("amount")         // DOUBLE
table.Decimal("price", 10, 2)  // DECIMAL with precision and scale

Boolean

table.Boolean("active")

Date/Time Types

table.Date("birth_date")
table.DateTime("created_at")
table.DateTime("created_at", 3) // with precision
table.DateTimeTz("created_at")    // with timezone
table.DateTimeTz("created_at", 3) // with timezone and precision
table.Time("start_time")
table.Time("start_time", 3)       // with precision
table.TimeTz("start_time")       // with timezone
table.TimeTz("start_time", 3)     // with timezone and precision
table.Timestamp("updated_at")
table.Timestamp("updated_at", 3)   // with precision
table.TimestampTz("updated_at")   // with timezone
table.TimestampTz("updated_at", 3) // with timezone and precision

JSON Types

table.JSON("metadata")     // JSON
table.Jsonb("data")         // JSONB (PostgreSQL)

Enum Type

table.Enum("status", []any{"active", "inactive", "pending"})

Geometry Types (Spatial)

table.Geometry("location")

table.Point("coordinates")

table.MultiPoint("points")
table.MultiLineString("paths")

Custom Column Type

table.Column("custom_type", "VARCHAR(255)")

Column Modifiers

Position Modifiers

table.String("name").After("id")      // Place after another column
table.String("name").First()          // Place as first column

Value Modifiers

table.String("name").Default("John")           // Set default value
table.String("name").Nullable()                   // Allow NULL values
table.Boolean("active").Default(true)
table.Timestamp("created_at").UseCurrent()          // Use current timestamp
table.Timestamp("updated_at").UseCurrentOnUpdate() // Update on record update

Index Modifiers

table.String("email").Unique()    // Add unique constraint
table.String("email").Index()     // Add index
table.String("email").Primary()   // Set as primary key

Other Modifiers

table.Integer("id").AutoIncrement()    // Auto-increment flag
table.String("name").Change()             // Mark column for modification
table.String("name").Collation("utf8mb4_unicode_ci") // Set collation
table.String("name").Comment("User name") // Add column comment
table.Decimal("price", 10, 2).Total(10).Places(2) // Decimal precision
table.Geometry("location").Srid(4326)     // Set SRID for geometry
table.Integer("age").Unsigned()          // Unsigned flag

Index Operations

// Regular index
table.Index("email")

// Composite index
table.Index("first_name", "last_name")

// Unique index
table.Unique("email")

// Fulltext index
table.FullText("content")

// Primary key
table.Primary("id")

// Foreign key
table.Foreign("user_id").References("id").On("users")

Drop Operations

// Drop table
table.Drop()

// Drop if exists
table.DropIfExists()

// Drop columns
table.DropColumn("old_column")
table.DropColumn("column1", "column2")

// Drop primary key
table.DropPrimary("id")

// Drop unique index
table.DropUnique("email")
table.DropUniqueByName("users_email_unique")

// Drop index
table.DropIndex("email")
table.DropIndexByName("users_email_index")

// Drop foreign key
table.DropForeign("user_id")
table.DropForeignByName("users_user_id_foreign")

// Drop fulltext index
table.DropFullText("content")
table.DropFullTextByName("posts_content_fulltext")

// Drop soft delete column
table.DropSoftDeletes(constants.DeletedAtColumnName)
table.DropSoftDeletesTz(constants.DeletedAtColumnName)

// Drop timestamps
table.DropTimestamps()
table.DropTimestampsTz()

Modification Operations

// Rename table
table.Rename("new_table_name")

// Rename column
table.RenameColumn("old_name", "new_name")

// Rename index
table.RenameIndex("old_index_name", "new_index_name")

// Mark for creation
table.Create()

// Mark column for modification
table.Change()

Soft Delete Columns

// NULL-based soft delete
table.SoftDeletes()              // soft_deleted_at (nullable)
table.SoftDeletes(constants.DeletedAtColumnName)  // deleted_at (nullable)

// Max-date sentinel soft delete (recommended for NOT NULL columns)
table.SoftDeletesMaxDate()              // soft_deleted_at with max-date default
table.SoftDeletesMaxDate(constants.DeletedAtColumnName)  // deleted_at with max-date default
table.SoftDeletesMaxDateTz()            // soft_deleted_tz with max-date default
table.SoftDeletesMaxDateTz(constants.DeletedAtColumnName) // deleted_tz with max-date default

// Or use the constant directly
import "github.com/dracory/neat/database/schema/constants"
table.Timestamp(constants.SoftDeleteAtColumn).Default(constants.MaxSoftDeletedAtDefault)

// With timezone (NULL-based)
table.SoftDeletesTz()           // soft_deleted_at with timezone
table.SoftDeletesTz("deleted_at") // deleted_at with timezone

Timestamp Helpers

// Standard timestamps
table.Timestamps()              // created_at, updated_at
table.Timestamps(3)             // with precision

// With timezone
table.TimestampsTz()            // created_at, updated_at with timezone
table.TimestampsTz(3)           // with timezone and precision

// Individual timestamps
table.CreatedAt()               // created_at only
table.UpdatedAt()               // updated_at only

Blueprint Utility Methods

// Get table name with prefix
tableName := table.GetTableName()

// Set table name
table.SetTable("custom_name")

// Check if command exists
if table.HasCommand("create") {
    // ...
}

// Get added columns
columns := table.GetAddedColumns()

// Get all commands
commands := table.GetCommands()

// Skip transaction
table.SkipTransaction()

// Check if should skip transaction
if table.ShouldSkipTransaction() {
    // ...
}

Column Modifiers

Unique

table.String("email").Unique()

Nullable

table.String("phone").Nullable()

Default

table.Boolean("active").Default(true)

Index

table.String("email").Index()

Indexes

// Single column index
table.Index("email")

// Composite index
table.Index(["first_name", "last_name"])

// Unique index
table.UniqueIndex("email")

Foreign Keys

table.ForeignKey("user_id").References("id").On("users")

Timestamps

table.Timestamps() // created_at and updated_at
table.SoftDeletes() // deleted_at

Checking Table/Column Existence

// Check if table exists
exists := db.Schema().HasTable("users")

// Check if column exists
hasColumn := db.Schema().HasColumn("users", "email")

Primary Key Types

Auto-Incrementing ID

db.Schema().Create("users", func(table neat.Blueprint) {
    table.ID() // BIGINT UNSIGNED AUTO_INCREMENT PRIMARY KEY
    table.String("name", 255)
    table.Timestamps()
})

Short String ID (Crockford Base32)

db.Schema().Create("products", func(table neat.Blueprint) {
    table.ShortID() // VARCHAR(21) PRIMARY KEY
    table.String("name", 255)
    table.Decimal("price", 10, 2)
    table.Timestamps()
})

Use with orm.ShortID embedded struct:

type Product struct {
    orm.ShortID
    orm.Timestamps
    Name  string  `db:"name"`
    Price float64 `db:"price"`
}
